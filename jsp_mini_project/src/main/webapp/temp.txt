signIn_process.jsp
<%@ page language="java" contentType="text/html; charset=UTF-8" pageEncoding="UTF-8"%>
<%@ page import="java.sql.*"%>
<%@ page import="javax.servlet.ServletException"%>
<%@ page import="java.net.URLEncoder"%>
<%@ include file="../db/dbconn.jsp"%>

<%
String userId = request.getParameter("userId");
String password = request.getParameter("pwd");
try {
    String sql = "SELECT role, password FROM ally_users WHERE userId = ?";
    PreparedStatement pstmt = conn.prepareStatement(sql);
    pstmt.setString(1, userId);

    ResultSet rs = pstmt.executeQuery();
    if (rs.next()) {
        String storedPassword = rs.getString("password");
        String userRole = rs.getString("role");

        boolean passwordMatch = password.equals(storedPassword);

        if (passwordMatch) {
            session.setAttribute("userId", userId);

            String redirectPage = "signIn_view.jsp"; // Default redirection
            String message = URLEncoder.encode(userId + "님, 환영합니다.", "UTF-8");

            if ("A".equalsIgnoreCase(userRole)) { // Admin role
                redirectPage = "userList.jsp";
            } else if ("B".equalsIgnoreCase(userRole)) { // Basic user
                redirectPage = "signIn_view.jsp"; // Or some other page for basic users
            }

            response.sendRedirect(redirectPage + "?message=" + message);
        } else {
            String message = URLEncoder.encode("Invalid username or password.", "UTF-8");
            response.sendRedirect("login.jsp?message=" + message); // Assuming 'login.jsp' is your login page
        }
    } else {
        String message = URLEncoder.encode("Invalid username or password.", "UTF-8");
        response.sendRedirect("login.jsp?message=" + message);
    }
} catch (SQLException e) {
    e.printStackTrace(); // For simplicity, just printing the stack trace. Consider logging.
    response.sendRedirect("login.jsp?message=" + URLEncoder.encode("Database error occurred.", "UTF-8"));
} catch (Exception e) {
    e.printStackTrace();
    response.sendRedirect("login.jsp?message=" + URLEncoder.encode("An error occurred.", "UTF-8"));
}
%>
